bool test_checksum(char* fname, ENTRY filed)
{
	std::fstream file;
	file.open(fname, std::fstream::in | std::fstream::binary);
	size_t res = 0;
	
	while(!file.eof())
	{
		res ^= file.get();
		res += file.get();
		res = ~res;
		res++;
	}
	
	file.close();
	
	if(res == filed.CHKSM)
		return true;
	return false;
}
